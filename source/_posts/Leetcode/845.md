### 解题思路

直接模拟该山的形状即可。

```java
class Solution {
    public int longestMountain(int[] A) {
        int len = A.length;
        if (len < 3) return 0;
        int ans = 0;
        int left = 0;
        int pre = left;
        int right = 1;
        while (right < len) {
            boolean up = false;
            boolean down = false;
            // walk through the flat road
            while (A[pre] == A[right]) {
                pre++;
                right++;
                if (right >= len) return ans;
            }
            left = pre;
            
            // walk up the road
            while (A[pre] < A[right]) {
                pre++;
                right++;
                up = true;
                if (right >= len) return ans;
            }
            
            // walk through the flat road
            while (A[pre] == A[right]) {
                pre++;
                right++;
                up = false;
                if (right >= len) return ans;
            }
            
            // walk down the road
            while (A[pre] > A[right]) {
                pre++;
                right++;
                down = true;
                if (right >= len) {
                    break;
                }
            }
            
            if (up && down) {
                ans = Math.max(ans, right - left);
                left = pre;
            }
            
        }
        return ans;
    }
}
```
